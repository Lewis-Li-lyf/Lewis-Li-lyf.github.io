介绍一下lowbit：
lowbit(i)lowbit(i)lowbit(i) 是指 2k2^k2k，其中 kkk 指 iii 用二进制表示时末尾0的个数，如8（1000）的kkk就是3，lowbit(8)lowbit(8)lowbit(8)就是23=82^3=823=8。

对于每一个C[i]C[ i ]C[i]，与之连接的下一个CCC为C[i+lowbit(i)]C[ i + lowbit(i) ]C[i+lowbit(i)]。

同样，求和公式：SUMi=C[i]+C[i−2k1]+C[(i−2k1)−2k2]+.....SUMi = C[i] + C[i-2^{k1}] + C[(i - 2^{k1}) - 2^{k2}] + .....SUMi=C[i]+C[i−2k1]+C[(i−2k1)−2k2]+.....

单点修改，区间查询

#include<cstdio>
using namespace std;

const int N=5*1e5;

int n,m,a[N+1],c[N+1],x,y,z;
int lowbit(int x){
	return x & (-x);
} 
void add(int pos,int k){
	while(pos <= n){
		c[pos] += k;
		pos += lowbit(pos);
	}
	return ;
}
int sum(int i){
	int res = 0;
	while(i > 0){
		res += c[i];
		i -= lowbit(i);
	}
	return res;
}
int main(){
	//freopen("putin.txt","r",stdin);
	//freopen("putout.txt","w",stdout);
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++){
		scanf("%d",&a[i]);
		add(i,a[i]);
	}
	for(int i=1;i<=m;i++){
		scanf("%d%d%d",&z,&x,&y);
		if(z == 1) add(x,y);
		if(z == 2){
			printf("%d\n",sum(y)-sum(x-1));//前缀和相减
		}
	}
	return 0;
}

区间修改，单点查询

#include<cstdio>
using namespace std;

const int N=5*1e5;

int n,m,a[N+1],c[N+1];
int lowbit(int x){
	return x & (-x);
} 
void add(int pos,int k){
	while(pos <= n){
		c[pos] += k;
		pos += lowbit(pos);
	}
	return ;
}
int getsum(int i){
	int res = 0;
	while(i > 0){
		res += c[i];
		i -= lowbit(i);
	}
	return res;
}
int main(){
	//freopen("putin.txt","r",stdin);
	//freopen("putout.txt","w",stdout);
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++){
		scanf("%d",&a[i]);
		add(i,a[i] - a[i-1]);//差分
	}
	for(int i=1;i<=m;i++){
		int z;
		scanf("%d",&z);
		if(z == 1){
			int x,y,k;
			scanf("%d%d%d",&x,&y,&k);
			add(x,k); 
			add(y+1,-k);//利用差分进行区间修改
		}
		if(z == 2){
			int s;
			scanf("%d",&s);
			printf("%d\n",getsum(s));
		}
	}
	return 0;
}
